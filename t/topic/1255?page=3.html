import os,sys,time
from libgen_uploader import LibgenUploader
from returns.pipeline import is_successful
from libgen_uploader import LibgenMetadata

u = LibgenUploader()
def main_cycle():time.sleep(1);main()
def main():
    try:
        lf=[]
        lf2=[]
        upurl=[]
        if os.path.exists('lf.txt'):
            f=open('lf.txt','r');lf=eval(f.read());f.close()
        if os.path.exists('lf2.txt'):
            f=open('lf2.txt','r');lf2=eval(f.read());f.close()
        
        for a,b,c in os.walk(sys.path[0]):
            for d in c:
                if d[-3:]!='jpg' and d[-2:]!='py' and d[-3:]!='txt' and d[-3:]!='swp':
                    print(lp:='%s/%s'%(a,d))
                    if lp not in lf:
                        m = LibgenMetadata(title='.'.join(lp.split('/')[-1:][0].split('.')[:-1]), language='Chinese')
                        result = u.upload_scitech(lp,metadata=m)
                        if is_successful(result):
                            upload_url = result.unwrap()
                            if lp not in lf:lf.append([lp])
                            if upload_url not in lf2:lf2.append([lp,upload_url])

                            f=open('lf.txt','w+')
                            for line in lf: {
                                f.write(line[0] + '\n')
                            }
                            f.close()

                            f=open('lf2.txt','w+')
                            for line in lf2: {
                                f.write(line[0] + '\n' + line[1] + '\n')
                            }
                            f.close()

                            print(len(lf),upload_url)
                        else:
                            failure = result.failure()
                            print(failure)
                            if lp not in lf:lf.append([lp])

                            f=open('lf.txt','w+')
                            for line in lf: {
                                f.write(line[0] + '\n')
                            }
                            f.close()
    except:main_cycle()
main()

106coup 2022 年8 月 13 日 05:18 #58

说实在的我不会建议去上传到 Z-Library，因为那就是一个盈利性的对 library genesis 的镜像，库存不公开，运营不透明。

书籍应当优先被传到 library genesis，LG 会把大家上传的书一千本一份全部公开出来，近两年来他们还为书籍做了 IPFS 功能。

由于 Z-Lib 是 LG 的镜像，所以用户上传到 LG 的所有书籍过一阵子都会出现在 Z-Lib。

LG 他们对书籍的质量比较重视，社区志愿的 Librarian 们会尽量去填充上传的每一本书的元数据。这种大量上传的行为他们比较反感（马列之声的这批资料连标题都参差不齐）。据管理员团队所言“他们手里待上传的书比已经在 LG 上的书还多，囤积资料不是重点，重点是填充尽可能多的元数据为人们所用”（We have more books to upload than are already on LG, hoarding material is not the point, the point is to fill in as much metadata as possible for people to use）。
