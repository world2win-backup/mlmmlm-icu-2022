def get_dash(avid, cid):
    urls = s.get(
        'https://api.bilibili.com/pgc/player/web/playurl?avid={}&cid={}&qn=0&fnver=0&fnval=4048&fourk=1'.format(avid,

        sys.exit("cannot get dash")

        return urls['result']['dash']
